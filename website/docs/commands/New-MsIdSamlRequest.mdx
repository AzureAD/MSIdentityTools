 [[-NameIDPolicyFormat] <String>] [[-RequestedAuthnContext] <String[]>]
 [[-RequestedAuthnContextComparison] <String>] [-DeflateAndEncode] [-UrlEncode]
 [-ProgressAction <ActionPreference>] [<CommonParameters>]
